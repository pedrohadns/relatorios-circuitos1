    \begin{multicols}{2}
        
        \begin{equation}
            P = \frac{dw}{dt}
            \label{eq: Potência}
        \end{equation}
        \columnbreak
        
        \noindent $P$ = Potência, em Watts\\
        $w$ = Trabalho, em Joules\\
        $t$ = Tempo, em segundos\\
    \end{multicols}
